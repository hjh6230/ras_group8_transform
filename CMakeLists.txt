cmake_minimum_required(VERSION 2.8.3)
project(ras_group8_transform)

find_package(catkin REQUIRED
  COMPONENTS
  roscpp
  phidgets
  tf
  sensor_msgs
  message_filters
  laser_geometry
)

catkin_package(
  INCLUDE_DIRS
    include

  CATKIN_DEPENDS
    roscpp
    phidgets
    tf
    sensor_msgs
    message_filters
    laser_geometry

# DEPENDS
)

include_directories(
  include
  ${catkin_INCLUDE_DIRS}
)

add_executable(${PROJECT_NAME}
  src/lidar_transform_node.cpp
  src/LidarTransform.cpp
)

add_dependencies(${PROJECT_NAME}
  ${catkin_EXPORTED_TARGETS}
)

target_link_libraries(${PROJECT_NAME}
  ${catkin_LIBRARIES}
)

#catkin_add_gtest(transform_tests test/test_transform.cpp src/Transform.cpp)

# Make main executable
install(
  TARGETS ${PROJECT_NAME}
  ARCHIVE DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
  LIBRARY DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
  RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
)

# Install headers
install(
  DIRECTORY include/${PROJECT_NAME}/
  DESTINATION ${CATKIN_PACKAGE_INCLUDE_DESTINATION}
  FILES_MATCHING PATTERN "*.hpp"
)


add_executable(scan2cloud src/ScantoCloud.cpp)
target_link_libraries(scan2cloud ${catkin_LIBRARIES})
